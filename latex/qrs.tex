\newcommand{\badge}[1]{

        \draw[thin,gray] (0,0) rectangle ++(5,7);
        \node at (2.5,4) { \qrcode[height=5cm]{#1} };
        \node[anchor=south east,font=\Huge\ttfamily] at (5,0) {#1};
    \end{tikzpicture}
}

    \badge{\x}
}
